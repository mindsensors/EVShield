49 inline uint32_t readLongFromBuffer(uint8_t* buf)
50 {
51  /* typecasts added to make it compatible with 1.6.8 */
52  return (uint32_t)buf[0] |
53  (((uint32_t)buf[1]) << 8) |
54  (((uint32_t)buf[2]) << 16) |
55  (((uint32_t)buf[3]) << 24);
56 }
